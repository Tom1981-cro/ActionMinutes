        {/* DEVICE FRAME */}

            {/* Mobile Header */}

/* 3. CAPTURE VIEW (Refined from Screenshot 3)                                */
/* ========================================================================== */
function CaptureView({ isMobile, startFocusTask }) {
  const [notes, setNotes] = useState(
    "# Weekly Sync: Marketing & Product\n\nDiscussed the Q3 budget constraints. We are slightly over in ad spend.\n\nAction: Sarah to send final adjusted numbers to the finance team by EOD.\n\nAlso need to upgrade staging servers before the new release goes out next week. \n\nTodo: Email IT about AWS limits (15m).\n\nGreat meeting overall. Momentum is good."
  );
  const [tasks, setTasks] = useState([]);
  const [showDetails, setShowDetails] = useState(false);

  useEffect(() => {
    const lines = notes.split('\n');
    const extracted = [];
    lines.forEach((line, index) => {
      const lower = line.toLowerCase();
      if (lower.includes('action:') || lower.includes('todo:')) {
        let cleanTitle = line.replace(/action:/i, '').replace(/todo:/i, '').trim();
        let minutes = 30; 
        const minMatch = cleanTitle.match(/\((\d+)m\)/);
        if (minMatch) {
          minutes = parseInt(minMatch[1], 10);
          cleanTitle = cleanTitle.replace(minMatch[0], '').trim();
        }
        extracted.push({ id: index, title: cleanTitle, minutes });
      }
    });
    setTasks(extracted);
  }, [notes]);

  return (
    <div className="p-6 h-full flex flex-col">
      <div className="flex justify-between items-center mb-6">
        <div>
          <h2 className="text-2xl font-bold text-gray-900">Capture</h2>
          <p className="text-sm text-gray-500 mt-1 font-medium">Record meetings and auto-extract actions.</p>
        </div>
      </div>

      <div className={`flex-1 flex ${isMobile ? 'flex-col space-y-6' : 'gap-6'} min-h-0`}>
        
        {/* Left Pane - Notepad (Yellowish tint) */}
        <div className="flex-1 flex flex-col bg-[#fffdf8] border border-amber-200 rounded-3xl overflow-hidden shadow-sm">
          <div className="bg-amber-50 px-6 py-5 border-b border-amber-100 flex flex-col space-y-4">
            <div className="flex justify-between items-start">
               <input type="text" placeholder="Meeting Title" defaultValue="Meeting — Feb 21" className="text-xl font-bold text-gray-900 bg-transparent outline-none w-full" />
               <span className="text-xs font-semibold text-amber-600 bg-amber-100/50 px-2.5 py-1 rounded ml-4 whitespace-nowrap border border-amber-200/50">21/02/2026</span>
            </div>
            
            {/* Toggleable Meeting Details */}
            {showDetails ? (
              <div className="grid grid-cols-2 gap-3 pb-2 pt-2 animate-in slide-in-from-top-2">
                <div className="col-span-2">
                  <input type="text" placeholder="Attendees (e.g. Alice, Bob)" className="w-full bg-white/80 border border-amber-200 rounded-lg px-3 py-2 text-sm outline-none focus:border-amber-400 placeholder-amber-700/40" />
                </div>
                <div>
                  <input type="text" placeholder="Time (--:--)" className="w-full bg-white/80 border border-amber-200 rounded-lg px-3 py-2 text-sm outline-none focus:border-amber-400 placeholder-amber-700/40" />
                </div>
                <div>
                  <input type="text" placeholder="Location" className="w-full bg-white/80 border border-amber-200 rounded-lg px-3 py-2 text-sm outline-none focus:border-amber-400 placeholder-amber-700/40" />
                </div>
                <button onClick={() => setShowDetails(false)} className="col-span-2 text-xs font-bold text-amber-600 hover:text-amber-800 text-left mt-1">Hide Details</button>
              </div>
            ) : (
              <button onClick={() => setShowDetails(true)} className="text-xs font-bold text-amber-600 hover:text-amber-800 text-left">Add attendees, time, location...</button>
            )}

            {/* Action Toolbar */}
            <div className="flex flex-wrap gap-2 pt-2 border-t border-amber-200/50">
               <button className="flex items-center text-[11px] font-bold text-white bg-violet-600 px-3 py-1.5 rounded-lg shadow-sm hover:bg-violet-700 transition-colors"><Mic className="w-3.5 h-3.5 mr-1.5"/> Record Meeting</button>
               <button className="flex items-center text-[11px] font-bold text-amber-800 bg-white border border-amber-200 px-3 py-1.5 rounded-lg shadow-sm hover:bg-amber-100 transition-colors"><ImageIcon className="w-3.5 h-3.5 mr-1.5"/> Upload Photo</button>
               <button className="flex items-center text-[11px] font-bold text-amber-800 bg-white border border-amber-200 px-3 py-1.5 rounded-lg shadow-sm hover:bg-amber-100 transition-colors"><Mic className="w-3.5 h-3.5 mr-1.5"/> Upload Audio</button>
            </div>
          </div>
          
          <textarea 
            className="flex-1 w-full p-6 text-[15px] text-gray-800 leading-relaxed font-medium bg-transparent border-none resize-none outline-none custom-scrollbar placeholder-gray-400"
            value={notes}
            onChange={(e) => setNotes(e.target.value)}
            placeholder="Start typing your meeting notes here...&#10;&#10;Prefix sentences with 'Action:' or 'Todo:' to automatically extract them."
          />
        </div>

        {/* Right Pane - Extraction */}
        <div className={`${isMobile ? 'w-full h-96' : 'w-[340px]'} bg-white rounded-3xl p-6 border border-gray-200 flex flex-col shadow-sm flex-shrink-0`}>
          <div className="flex justify-between items-center mb-6 border-b border-gray-100 pb-4">
            <h3 className="text-xs font-bold text-gray-500 uppercase tracking-widest flex items-center">
              <Target className="w-4 h-4 mr-2 text-violet-500" /> Extracted Actions

              {tasks.length} Found

          ) : (
            <ul className="space-y-3 flex-1 overflow-y-auto custom-scrollbar pr-1">

                  <span className="text-sm font-semibold text-gray-800 mb-2">{task.title}</span>
                  <div className="flex items-center space-x-2">
                    <span className="text-[10px] font-bold bg-white border border-gray-200 text-gray-500 px-2 py-1 rounded group-hover:text-violet-600 transition-colors">
                      {task.minutes}m allocated
                    </span>
                  </div>
                  <button 
                    onClick={() => startFocusTask(task.title)} 
                    className="absolute right-3 bottom-3 p-1.5 bg-violet-100 text-violet-600 rounded-lg opacity-0 group-hover:opacity-100 transition-opacity shadow-sm"
                    title="Start Focus Ring"
                  >
                    <Play className="w-3.5 h-3.5" />
                  </button>
                </li>
              ))}
            </ul>
          )}
          
          <button className="w-full mt-6 bg-violet-600 text-white py-3.5 rounded-xl font-bold text-sm shadow-md hover:bg-violet-700 transition-colors flex items-center justify-center">
            <CheckCircle className="w-4 h-4 mr-2" /> Import Actions
          </button>
        </div>
      </div>
    </div>
  );
}

                        className="absolute right-2 top-2 p-1.5 bg-violet-100 text-violet-600 rounded-lg opacity-0 group-hover:opacity-100 transition-opacity shadow-sm"
