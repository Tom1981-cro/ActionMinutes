    <script>
        // --- Shared Drawing Function ---
        function drawLogo(canvasId, options = {}) {
            const canvas = document.getElementById(canvasId);
            if (!canvas) return;
            const ctx = canvas.getContext('2d');
            const w = canvas.width;
            const h = canvas.height;
            const cx = w / 2;
            const cy = h / 2;

            // Options
            const bgColor = options.bgColor || '#14b8a6'; // Teal-500
            const iconColor = options.iconColor || '#ffffff';
            const textColor = options.textColor || '#1e293b'; // Slate-800
            const secondaryTextColor = options.secondaryTextColor || '#475569'; // Slate-600
            const shape = options.shape || 'squircle'; // squircle, circle, none
            const withText = options.withText || false;
            const scale = options.scale || 1;

            // Clear
            ctx.clearRect(0, 0, w, h);

            // 1. Draw Background Shape
            if (shape !== 'none') {
                ctx.fillStyle = bgColor;
                if (shape === 'circle') {
                    ctx.beginPath();
                    ctx.arc(cx, cy, w/2, 0, Math.PI * 2);
                    ctx.fill();
                } else if (shape === 'squircle') {
                    // Smooth rounded rect (Squircle approximation)
                    const r = w * 0.35; // Radius
                    ctx.beginPath();
                    ctx.roundRect(0, 0, w, h, r);
                    ctx.fill();
                } else if (shape === 'rounded-header') {
                    // Small icon for header
                    const r = w * 0.35;
                    ctx.beginPath();
                    ctx.roundRect(0, 0, w, h, r);
                    ctx.fill();
                }
            }

            // 2. Draw The Symbol (The "Time-Check")
            // Concept: A clock face where the hands form a check mark.
            // Center point of the check depends on canvas size
            
            // If Text is present, offset the icon to the left
            let iconX = cx;
            let iconY = cy;
            let iconSize = w * 0.6; // Base size relative to container

            if (withText) {
                iconSize = h * 0.8; // Icon matches height
                iconX = iconSize / 2; // Left aligned
                iconY = h / 2;
                
                // Draw Icon Container for Header
                ctx.fillStyle = bgColor;
                ctx.beginPath();
                ctx.roundRect(0, 0, iconSize, iconSize, iconSize * 0.35);
                ctx.fill();
                
                // Update center for drawing stroke
                iconX = iconSize / 2;
            }

            ctx.strokeStyle = iconColor;
            ctx.lineWidth = iconSize * 0.12; // Thick stroke
            ctx.lineCap = 'round';
            ctx.lineJoin = 'round';

            // Draw Checkmark
            // Start (left short stroke) -> Center (bottom) -> End (right long stroke)
            // Coordinates relative to icon center
            const checkStart = { x: iconX - (iconSize * 0.2), y: iconY + (iconSize * 0.05) };
            const checkPivot = { x: iconX - (iconSize * 0.05), y: iconY + (iconSize * 0.25) };
            const checkEnd = { x: iconX + (iconSize * 0.25), y: iconY - (iconSize * 0.25) };

            ctx.beginPath();
            ctx.moveTo(checkStart.x, checkStart.y);
            ctx.lineTo(checkPivot.x, checkPivot.y);
            ctx.lineTo(checkEnd.x, checkEnd.y);
            ctx.stroke();

            // Draw "Clock Ring" Hint (Optional subtle detail)
            // A small arc to suggest time
            ctx.globalAlpha = 0.3;
            ctx.lineWidth = iconSize * 0.05;
            ctx.beginPath();
            ctx.arc(iconX, iconY, iconSize * 0.35, Math.PI * 1.2, Math.PI * 2.2);
            ctx.stroke();
            ctx.globalAlpha = 1.0;

            // 3. Draw Text (if enabled)
            if (withText) {
                const textSize = h * 0.6;
                const textX = iconSize * 1.3; // Padding
                const textY = cy + (textSize * 0.35);

                ctx.font = `bold ${textSize}px Outfit`;
                ctx.fillStyle = textColor;
                ctx.fillText("Action", textX, textY);
                
                const actionWidth = ctx.measureText("Action").width;
                ctx.font = `400 ${textSize}px Outfit`;
                ctx.fillStyle = secondaryTextColor;
                ctx.fillText("Minutes", textX + actionWidth, textY);
            }
        }

        // --- Render All Scenarios ---
        function renderAll() {
            // 1. Navigation Logo
            drawLogo('nav-logo', { shape: 'squircle', scale: 0.1 });

            // 2. Hero Large Logo
            drawLogo('hero-logo', { shape: 'squircle' });

            // 3. Web Header
            drawLogo('scenario-header', { 
                shape: 'none', // Custom handling inside
                withText: true 
            });

            // 4. Mobile App (Full Bleed)
            drawLogo('scenario-app', { shape: 'squircle' });

            // 5. Social Avatar (Circle)
            drawLogo('scenario-social', { shape: 'circle' });

            // 6. Dark Mode
            drawLogo('scenario-dark', { 
                shape: 'none', 
                withText: true,
                bgColor: '#2dd4bf', // Brighter teal for dark mode
                textColor: '#ffffff',
                secondaryTextColor: '#94a3b8' // Slate-400
            });
        }

        // --- Context Switching Logic ---
        function setContext(mode) {
            const wrapper = document.querySelector('.canvas-wrapper');
            const contextLabel = wrapper.querySelector('div'); // The label

            if (mode === 'light') {
                wrapper.style.backgroundColor = '#ffffff';
                wrapper.style.backgroundImage = 'radial-gradient(#e7e5e4 1px, transparent 1px)';
                drawLogo('hero-logo', { shape: 'squircle', bgColor: '#14b8a6', iconColor: '#fff' });
                contextLabel.style.color = '#78716c';
            } else if (mode === 'dark') {
                wrapper.style.backgroundColor = '#0f172a'; // Slate-900
                wrapper.style.backgroundImage = 'radial-gradient(#334155 1px, transparent 1px)';
                drawLogo('hero-logo', { shape: 'squircle', bgColor: '#2dd4bf', iconColor: '#0f172a' }); // High contrast
                contextLabel.style.color = '#94a3b8';
            } else if (mode === 'brand') {
                wrapper.style.backgroundColor = '#14b8a6'; // Teal-500
                wrapper.style.backgroundImage = 'none';
                drawLogo('hero-logo', { shape: 'squircle', bgColor: '#ffffff', iconColor: '#14b8a6' }); // Inverted
                contextLabel.style.color = '#ccfbf1';
            }
        }

        // Init
        document.addEventListener('DOMContentLoaded', () => {
            renderAll();
        });
